005package com.nuecho.rivr.core.channel.synchronous.step;
006
007import com.nuecho.rivr.core.channel.*;
008import com.nuecho.rivr.core.util.*;
009
010/**
011 * Step wrapping a {@link LastTurn}. This kind of step tells that the dialogue
012 * is done.
013 *
014 * @param <O> type of {@link OutputTurn}
015 * @param <L> type of {@link LastTurn}
016 * @author Nu Echo Inc.
017 */
018public final class LastTurnStep<O extends OutputTurn, L extends LastTurn> implements Step<O, L> {
019
020    private final L mLastTurn;
021
022    public LastTurnStep(L lastTurn) {
023        Assert.notNull(lastTurn, "lastTurn");
024        mLastTurn = lastTurn;
025    }
026
027    public L getLastTurn() {
028        return mLastTurn;
029    }
030
031    @Override
032    public String toString() {
033        return "Last Turn: " + mLastTurn;
034    }
035
036    @Override
037    public int hashCode() {
038        final int prime = 31;
039        int result = 1;
040        result = prime * result + ((mLastTurn == null) ? 0 : mLastTurn.hashCode());
041        return result;
042    }
043
044    @Override
045    public boolean equals(Object obj) {
046        if (this == obj) return true;
047        if (obj == null) return false;
048        if (getClass() != obj.getClass()) return false;
049        LastTurnStep other = (LastTurnStep) obj;
050        if (mLastTurn == null) {
051            if (other.mLastTurn != null) return false;
052        } else if (!mLastTurn.equals(other.mLastTurn)) return false;
053        return true;
054    }
055}
