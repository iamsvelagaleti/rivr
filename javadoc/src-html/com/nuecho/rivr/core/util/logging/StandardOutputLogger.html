005package com.nuecho.rivr.core.util.logging;
006
007import java.io.*;
008
009import org.slf4j.*;
010
011/**
012 * Standard Output Console {@link Logger}
013 * 
014 * @author Nu Echo Inc.
015 */
016public class StandardOutputLogger extends ConsoleLogger {
017
018    public StandardOutputLogger(LogLevel logLevel) {
019        super(logLevel);
020    }
021
022    @Override
023    protected PrintStream getPrintStream() {
024        return System.out;
025    }
026
027    @Override
028    public String getName() {
029        return "Standard Output";
030    }
031
032}
