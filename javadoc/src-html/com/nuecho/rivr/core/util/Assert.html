005package com.nuecho.rivr.core.util;
006
007import static java.util.Arrays.*;
008
009import java.util.*;
010
011/**
012 * Assertion facilities.
013 * 
014 * @author Nu Echo Inc.
015 */
016public final class Assert {
017
018    private Assert() {
019        //utility class: instantiation forbidden
020    }
021
022    public static void notNull(Object item, String symbol) {
023        if (item == null) throw new AssertionError(symbol + " should not be null");
024    }
025
026    public static void notEmpty(String string, String symbol) {
027        notNull(string, symbol);
028        ensure(!string.isEmpty(), symbol + " should not be empty.");
029    }
030
031    public static void notEmpty(Collection<?> collection, String symbol) {
032        notNull(collection, symbol);
033        ensure(!collection.isEmpty(), symbol + " should not be empty.");
034    }
035
036    public static void noNullValues(Collection<?> collection, String symbol) {
037        Assert.notNull(collection, symbol);
038        for (Object item : collection) {
039            ensure(item != null, symbol + " should not contain null values.");
040        }
041    }
042
043    public static void noNullValues(Object[] array, String symbol) {
044        Assert.notNull(array, symbol);
045        int index = 0;
046        for (Object item : array) {
047            ensure(item != null, symbol + "[" + index + "]  should not be null.");
048            index++;
049        }
050    }
051
052    public static void noNullValues(List<?> list, String symbol) {
053        Assert.notNull(list, symbol);
054        int index = 0;
055        for (Object item : list) {
056            ensure(item != null, "item #" + index + " of " + symbol + " should not be null.");
057            index++;
058        }
059    }
060
061    public static void between(long min, long value, long max) {
062        if (value < min || value > max)
063            throw new AssertionError("Value should not be in the range [" + min + ", " + max + "]");
064    }
065
066    public static void between(double min, double value, double max) {
067        if (value < min || value > max)
068            throw new AssertionError("Value should not be in the range [" + min + ", " + max + "]");
069    }
070
071    public static void notNegative(long value, String symbol) {
072        ensure(value >= 0, symbol + " should not be negative. Value=" + value);
073    }
074
075    public static void positive(long value, String symbol) {
076        ensure(value > 0, symbol + " should be positive. Value=" + value);
077    }
078
079    public static void ensure(boolean condition, String message) {
080        if (!condition) throw new AssertionError(message);
081    }
082
083    public static <T> List<T> asListChecked(T[] items) {
084        notNull(items, "items");
085        return asList(items);
086    }
087
088}
