005package com.nuecho.rivr.core.util;
006
007import java.io.*;
008import java.util.*;
009
010/**
011 * @author Nu Echo Inc.
012 * @since 1.0.3
013 */
014public class VersionBase {
015
016    private static final String VERSION_PROPERTY = "version";
017    private static final String VERSION_PROPERTIES_RESOURCE_NAME = "version.properties";
018
019    protected String getVersion(Class<? extends VersionBase> versionClass) {
020        try {
021            Properties versionProperties = new Properties();
022            InputStream versionPropertyFile = versionClass.getResourceAsStream(VERSION_PROPERTIES_RESOURCE_NAME);
023            versionProperties.load(versionPropertyFile);
024            String version = versionProperties.getProperty(VERSION_PROPERTY);
025            return version == null ? "" : version;
026        } catch (IOException exception) {
027            return "";
028        }
029    }
030
031}
