206        vxmlElement.setAttributeNS(XMLConstants.XMLNS_ATTRIBUTE_NS_URI,

215    public static Document createDocument(VoiceXmlDialogueContext voiceXmlDialogueContext) {
216        Element vxmlElement = createVoiceXmlDocumentRoot(voiceXmlDialogueContext);
217
218        String rootDocumentPath = voiceXmlDialogueContext.getContextPath()
219                                  + voiceXmlDialogueContext.getServletPath()
220                                  + VoiceXmlDialogueServlet.ROOT_PATH
221                                  + voiceXmlDialogueContext.getDialogueId();
222
223        vxmlElement.setAttribute(APPLICATION_ATTRIBUTE, rootDocumentPath);
224
225        StringBuilder script = new StringBuilder();
226        script.append(RIVR_SCOPE_OBJECT)
227              .append(".")
228              .append(LOCAL_ERROR_HANDLING_PROPERTY)
229              .append(" = ")
230              .append(FALSE)
231              .append("; ");
232
233        script.append(RIVR_INPUT_TURN_SCOPE_OBJECT).append(" = {};");
234        createScript(vxmlElement, script.toString());
235
236        processFetchRendering(voiceXmlDialogueContext, vxmlElement);
237
238        processProperties(voiceXmlDialogueContext, vxmlElement);
239
240        return vxmlElement.getOwnerDocument();
241    }

252    private static void processFetchRendering(VoiceXmlDialogueContext voiceXmlDialogueContext, Element vxmlElement) {
253        DefaultFetchConfiguration defaultFetchConfiguration = voiceXmlDialogueContext.getFetchConfiguration();
254        if (defaultFetchConfiguration == null) return;
255
256        String fetchAudioLocation = defaultFetchConfiguration.getDefaultFetchAudio();
257        if (fetchAudioLocation != null) {
258            addFetchAudioProperty(vxmlElement, fetchAudioLocation);
259        }
260
261        addDurationProperty(vxmlElement, FETCH_TIMEOUT_PROPERTY, defaultFetchConfiguration.getDefaultFetchTimeout());
262
263        applyResourceDefaultFetchConfiguration(vxmlElement,
264                                               defaultFetchConfiguration.getDefaultAudioFetchConfiguration());
265        applyResourceDefaultFetchConfiguration(vxmlElement,
266                                               defaultFetchConfiguration.getDefaultGrammarFetchConfiguration());
267        applyResourceDefaultFetchConfiguration(vxmlElement,
268                                               defaultFetchConfiguration.getDefaultObjectFetchConfiguration());
269        applyResourceDefaultFetchConfiguration(vxmlElement,
270                                               defaultFetchConfiguration.getDefaultScriptFetchConfiguration());
271    }

302    public static void createPrompt(String language,
303                                    Element parent,
304                                    VoiceXmlDialogueContext voiceXmlDialogueContext,
305                                    Boolean bargeIn,
306                                    List<? extends AudioItem> audioItems) throws VoiceXmlDocumentRenderingException {
307
308        Element promptElement = DomUtils.appendNewElement(parent, PROMPT_ATTRIBUTE);
309        setBooleanAttribute(promptElement, BARGE_IN_PROPERTY, bargeIn);
310
311        String documentLanguage = voiceXmlDialogueContext.getLanguage();
312        if (language != null && !language.equals(documentLanguage)) {
313            promptElement.setAttribute(XML_LANGUAGE_ATTRIBUTE, language);
314        }
315
316        processAudioItems(audioItems, promptElement);
317    }

319    public static void processAudioItems(List<? extends AudioItem> audioItems, Element promptElement)
320            throws VoiceXmlDocumentRenderingException {
321
322        boolean lastItemWasText = false;
323        for (AudioItem audioItem : audioItems) {
324            if (audioItem instanceof AudioFile) {
325                AudioFile audioFile = (AudioFile) audioItem;
326                Element audioElement = DomUtils.appendNewElement(promptElement, AUDIO_ELEMENT);
327
328                //src and expr are mutually exclusive, this is enforced in AudioFile.
329                setAttribute(audioElement, SRC_ATTRIBUTE, audioFile.getLocation());
330                setAttribute(audioElement, EXPR_ATTRIBUTE, audioFile.getExpression());
331
332                SpeechSynthesis alternate = audioFile.getAlternate();
333
334                if (alternate != null) {
335                    if (alternate.isSsml()) {
336                        DocumentFragment ssmlNodes = alternate.getDocumentFragment();
337                        audioElement.appendChild(audioElement.getOwnerDocument().importNode(ssmlNodes, true));
338                        lastItemWasText = false;
339                    } else {
340                        DomUtils.appendNewText(audioElement, alternate.getText());
341                        lastItemWasText = true;
342                    }
343                }
344
345                applyRessourceFetchConfiguration(audioElement, audioFile.getFetchConfiguration());
346            } else if (audioItem instanceof SpeechSynthesis) {
347                SpeechSynthesis speechSynthesis = (SpeechSynthesis) audioItem;
348
349                if (speechSynthesis.isSsml()) {
350                    NodeList childNodes = speechSynthesis.getDocumentFragment().getChildNodes();
351                    Document document = promptElement.getOwnerDocument();
352
353                    for (int childNodeIndex = 0; childNodeIndex < childNodes.getLength(); childNodeIndex++) {
354                        Node node = document.importNode(childNodes.item(childNodeIndex).cloneNode(true), true);
355                        promptElement.appendChild(node);
356                    }
357                    lastItemWasText = false;
358                } else {
359                    String spaceIfRequired = lastItemWasText ? " " : "";
360                    DomUtils.appendNewText(promptElement, spaceIfRequired + speechSynthesis.getText());
361                    lastItemWasText = true;
362                }
363            } else if (audioItem instanceof Pause) {
364                Pause pause = (Pause) audioItem;
365                Element breakElement = DomUtils.appendNewElement(promptElement, BREAK_ELEMENT);
366                setDurationAttribute(breakElement, TIME_ATTRIBUTE, pause.getDuration());
367                lastItemWasText = false;
368            } else if (audioItem instanceof Mark) {
369                Mark mark = (Mark) audioItem;
370                Element markElement = DomUtils.appendNewElement(promptElement, MARK_ELEMENT);
371                markElement.setAttribute(NAME_ATTRIBUTE, mark.getName());
372                lastItemWasText = false;
373            } else throw new VoiceXmlDocumentRenderingException("Cannot handle prompt rendering element of type '"
374                                                                + audioItem.getClass()
375                                                                + "'");
376        }
377    }

453    public static String createEcmaScriptStringLiteral(String content) {
454        if (content == null) return "undefined";
455
456        //believe it: this creates \\ for each \
457        content = ECMASCRIPT_BACKSLASH_PATTERN.matcher(content).replaceAll("\\\\\\\\");
458
459        content = ECMASCRIPT_QUOTE_PATTERN.matcher(content).replaceAll("\\\\'");
460        content = ECMASCRIPT_LINEFEED_PATTERN.matcher(content).replaceAll("\\u000a");
461        content = ECMASCRIPT_CARRIAGE_RETURN_PATTERN.matcher(content).replaceAll("\\u000d");
462        content = ECMASCRIPT_LINE_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2028");
463        content = ECMASCRIPT_PARAGRAPH_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2029");
464
465        StringBuilder out = new StringBuilder();
466        out.append("'").append(content).append("'");
467        return out.toString();
468    }

640    public static void renderGrammars(List<? extends GrammarItem> grammarItems, Element parent, String mode)
641            throws VoiceXmlDocumentRenderingException {
642        for (GrammarItem grammarItem : grammarItems) {
643            Element grammarElement;
644
645            if (grammarItem instanceof GrammarReference) {
646                grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
647                GrammarReference grammarReference = (GrammarReference) grammarItem;
648
649                VoiceXmlDomUtil.renderGrammarReference(grammarReference, grammarElement);
650            } else if (grammarItem instanceof InlineStringGrammar) {
651                grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
652                InlineStringGrammar inlineStringGrammar = (InlineStringGrammar) grammarItem;
653                VoiceXmlDomUtil.renderInlineStringGrammar(grammarElement, inlineStringGrammar);
654            } else if (grammarItem instanceof InlineXmlGrammar) {
655                InlineXmlGrammar inlineXmlGrammar = (InlineXmlGrammar) grammarItem;
656                Document document = inlineXmlGrammar.getDocument();
657                Element documentRootElement = document.getDocumentElement();
658                if (!documentRootElement.getTagName().equals(GRAMMAR_ELEMENT))
659                    throw new VoiceXmlDocumentRenderingException("XML grammar root element must be 'gramar'");
660
661                grammarElement = (Element) parent.getOwnerDocument().importNode(documentRootElement, true);
662                parent.appendChild(grammarElement);
663            } else throw new AssertionError("Unsupported grammar item: " + grammarItem);
664
665            Double weight = grammarItem.getWeight();
666            if (weight != null) {
667                DecimalFormat weigthDecimalFormat = new DecimalFormat("#.##########");
668                double weightDoubleValue = weight.doubleValue();
669                grammarElement.setAttribute(WEIGHT_ATTRIBUTE, weigthDecimalFormat.format(weightDoubleValue));
670            }
671
672            setAttribute(grammarElement, TYPE_ATTRIBUTE, grammarItem.getMediaType());
673            setAttribute(grammarElement, MODE_ATTRIBUTE, mode);
674        }
675    }

708    public static void addNamelist(Element parentElement, Element namelistHolderElement, VariableList variables) {
709        if (variables == null) return;
710        if (variables.isEmpty()) return;
711
712        List<String> variableNames = new ArrayList<String>();
713        for (Entry<String, String> entry : variables) {
714            variableNames.add(entry.getKey());
715            createVarElement(parentElement, entry.getKey(), entry.getValue());
716        }
717
718        namelistHolderElement.setAttribute(NAME_LIST_ATTRIBUTE, StringUtils.join(variableNames, " "));
719    }
