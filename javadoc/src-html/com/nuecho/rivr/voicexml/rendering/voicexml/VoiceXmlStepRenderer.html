007import java.io.*;
008import java.util.*;
009
010import javax.servlet.http.*;
011
012import org.w3c.dom.*;

014import com.nuecho.rivr.core.servlet.*;
015import com.nuecho.rivr.core.util.*;

017import com.nuecho.rivr.voicexml.turn.*;

019import com.nuecho.rivr.voicexml.turn.last.*;

022/**
023 * VoiceXML specialization of {@link StepRenderer}. {@link VoiceXmlOutputTurn
024 * VoiceXmlOutputTurns} and {@link VoiceXmlLastTurn VoiceXmlLastTurns} are
025 * converted to VoiceXML documents.
026 * 
027 * @author Nu Echo Inc.
028 */
029public class VoiceXmlStepRenderer implements
030        StepRenderer<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> {
031
032    private static final String VOICE_XML_MIME_TYPE = "application/voicexml+xml";
033
034    private final List<VoiceXmlDocumentAdapter> mVoiceXmlDocumentAdapters;
035
036    public VoiceXmlStepRenderer(List<? extends VoiceXmlDocumentAdapter> voiceXmlDocumentAdapters) {
037        if (voiceXmlDocumentAdapters != null) {
038            mVoiceXmlDocumentAdapters = new ArrayList<VoiceXmlDocumentAdapter>(voiceXmlDocumentAdapters);
039        } else {
040            mVoiceXmlDocumentAdapters = null;
041        }
042    }
043
044    @Override
045    public ServletResponseContent createDocumentForOutputTurn(VoiceXmlOutputTurn outputTurn,
046                                                              HttpServletRequest request,
047                                                              HttpServletResponse response,
048                                                              VoiceXmlDialogueContext voiceXmlDialogueContext)
049            throws StepRendererException {
050        return renderTurn(outputTurn, voiceXmlDialogueContext);
051    }
052
053    @Override
054    public ServletResponseContent createDocumentForLastTurn(VoiceXmlLastTurn lastTurn,
055                                                            HttpServletRequest request,
056                                                            HttpServletResponse response,
057                                                            VoiceXmlDialogueContext voiceXmlDialogueContext)
058            throws StepRendererException {
059        return renderTurn(lastTurn, voiceXmlDialogueContext);
060    }
061
062    private ServletResponseContent renderTurn(VoiceXmlDocumentTurn turn, VoiceXmlDialogueContext voiceXmlDialogueContext)
063            throws StepRendererException {
064        Assert.notNull(turn, "turn");
065        Document voiceXmlDocument;
066        try {
067            voiceXmlDocument = turn.getVoiceXmlDocument(voiceXmlDialogueContext);
068        } catch (VoiceXmlDocumentRenderingException exception) {
069            throw new StepRendererException(exception);
070        }
071
072        try {
073            if (mVoiceXmlDocumentAdapters != null) {
074                for (VoiceXmlDocumentAdapter adapter : mVoiceXmlDocumentAdapters) {
075                    adapter.adaptVoiceXmlDocument(voiceXmlDocument);
076                }
077            }
078        } catch (VoiceXmlDocumentRenderingException exception) {
079            throw new StepRendererException("Error white applying adapter.", exception);
080        }
081
082        voiceXmlDialogueContext.incrementTurnIndex();
083        try {
084            return new XmlDocumentServletResponseContent(voiceXmlDocument, VOICE_XML_MIME_TYPE);
085        } catch (IOException exception) {
086            throw new StepRendererException("Unable to create response.", exception);
087        }
088    }
089
090}
