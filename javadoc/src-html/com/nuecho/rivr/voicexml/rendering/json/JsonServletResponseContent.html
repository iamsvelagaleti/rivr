005package com.nuecho.rivr.voicexml.rendering.json;
006
007import java.io.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.core.servlet.*;
012import com.nuecho.rivr.core.util.*;
013import com.nuecho.rivr.voicexml.util.json.*;
014
015/**
016 * Wraps a JSON value in a {@link ServletResponseContent}.
017 * 
018 * @author Nu Echo Inc.
019 */
020public class JsonServletResponseContent implements ServletResponseContent {
021
022    private static final String JSON_MIME_TYPE = "application/json";
023    private static final String HTML_MIME_TYPE = "text/html";
024
025    private final byte[] mContent;
026    private final JsonpMode mJsonpMode;
027
028    enum JsonpMode {
029        DISABLED, NORMAL, TEXTAREA
030    }
031
032    public JsonServletResponseContent(JsonStructure jsonData, JsonpMode jsonpMode, String jsonpCallback)
033            throws IOException {
034        Assert.notNull(jsonpMode, "jsonpMode");
035        Assert.notNull(jsonData, "jsonData");
036
037        mJsonpMode = jsonpMode;
038        mContent = buildContent(jsonData, jsonpCallback);
039    }
040
041    @Override
042    public String getContentType() {
043        if (mJsonpMode == JsonpMode.DISABLED) return JSON_MIME_TYPE;
044        else if (mJsonpMode == JsonpMode.NORMAL) return JSON_MIME_TYPE;
045        else if (mJsonpMode == JsonpMode.TEXTAREA) return HTML_MIME_TYPE;
046        else throw new AssertionError("Unexpected JSONP mode: " + mJsonpMode);
047    }
048
049    @Override
050    public void writeTo(OutputStream outputStream) throws IOException {
051        outputStream.write(mContent);
052    }
053
054    @Override
055    public String getContentAsString() {
056        try {
057            return new String(mContent, Encoding.UTF_8.getId());
058        } catch (UnsupportedEncodingException exception) {
059            throw new AssertionError("Missing " + Encoding.UTF_8.getId() + " encoding.");
060        }
061    }
062
063    private byte[] buildContent(JsonStructure jsonData, String jsonpCallback) throws IOException {
064
065        ByteArrayOutputStream binaryArrayOutputStream = new ByteArrayOutputStream();
066        Writer writer = new OutputStreamWriter(binaryArrayOutputStream, Encoding.UTF_8.getId());
067
068        if (mJsonpMode == JsonpMode.TEXTAREA) {
069            writer.write("<textarea>");
070        }
071
072        if (mJsonpMode != JsonpMode.DISABLED) {
073            writer.write(jsonpCallback);
074            writer.write('(');
075        }
076
077        JsonUtils.write(writer, jsonData);
078
079        if (mJsonpMode != JsonpMode.DISABLED) {
080            writer.write(')');
081        }
082
083        if (mJsonpMode == JsonpMode.TEXTAREA) {
084            writer.write("</textarea>");
085        }
086
087        writer.flush();
088        return binaryArrayOutputStream.toByteArray();
089    }
090
091    /**
092     * @since 1.0.2
093     */
094    @Override
095    public Integer getContentLength() {
096        return mContent.length;
097    }
098
099}
