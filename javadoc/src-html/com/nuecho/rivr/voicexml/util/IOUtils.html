005package com.nuecho.rivr.voicexml.util;
006
007import java.io.*;
008
009import com.nuecho.rivr.core.util.*;
010
011/**
012 * IO utility class.
013 * 
014 * @author Nu Echo Inc.
015 */
016public final class IOUtils {
017
018    private static final int BUFFER_SIZE = 4096;
019
020    public static byte[] toByteArray(InputStream inputStream) throws IOException {
021        Assert.notNull(inputStream, "inputStream");
022        byte[] buffer = new byte[BUFFER_SIZE];
023        InputStream bufferedInputStream = new BufferedInputStream(inputStream, BUFFER_SIZE);
024        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
025
026        int length = 0;
027        while ((length = bufferedInputStream.read(buffer, 0, BUFFER_SIZE)) != -1) {
028            outputStream.write(buffer, 0, length);
029        }
030
031        return outputStream.toByteArray();
032    }
033
034    private IOUtils() {}
035}
