005package com.nuecho.rivr.voicexml.util;
006
007import javax.json.*;
008
009import com.nuecho.rivr.voicexml.turn.last.*;
010import com.nuecho.rivr.voicexml.util.json.*;
011
012/**
013 * Utility class containing helper methods for result returned in some
014 * {@link VoiceXmlLastTurn last turns}.
015 * 
016 * @author Nu Echo Inc.
017 */
018public class ResultUtils {
019
020    private static final String STACK_TRACE_PROPERTY = "stackTrace";
021    private static final String CLASS_PROPERTY = "class";
022    private static final String MESSAGE_PROPERTY = "message";
023
024    public static final JsonValue toJson(Throwable throwable) {
025        JsonArrayBuilder builder = JsonUtils.createArrayBuilder();
026        addThrowable(builder, throwable, null);
027        return builder.build();
028    }
029
030    private static void addThrowable(JsonArrayBuilder arrayBuilder, Throwable throwable, Throwable parent) {
031        JsonObjectBuilder topLevelObjectBuilder = JsonUtils.createObjectBuilder();
032        String message = throwable.getMessage();
033        if (message != null) {
034            JsonUtils.add(topLevelObjectBuilder, MESSAGE_PROPERTY, message);
035        }
036
037        JsonUtils.add(topLevelObjectBuilder, CLASS_PROPERTY, throwable.getClass().getName());
038
039        StackTraceElement[] stackTraceElements = throwable.getStackTrace();
040        JsonArrayBuilder stackTraceBuilder = JsonUtils.createArrayBuilder();
041
042        int framesInCommon;
043        if (parent != null) {
044            StackTraceElement[] thisStackTrace = stackTraceElements;
045            StackTraceElement[] parentStackTrace = parent.getStackTrace();
046
047            int thisStackTraceIndex = thisStackTrace.length - 1;
048            int otherStackTraceIndex = parentStackTrace.length - 1;
049            while (thisStackTraceIndex >= 0
050                   && otherStackTraceIndex >= 0
051                   && thisStackTrace[thisStackTraceIndex].equals(parentStackTrace[otherStackTraceIndex])) {
052                thisStackTraceIndex--;
053                otherStackTraceIndex--;
054            }
055            framesInCommon = thisStackTrace.length - 1 - thisStackTraceIndex;
056        } else {
057            framesInCommon = 0;
058        }
059
060        int maxFrameIndex = stackTraceElements.length - framesInCommon;
061
062        for (int index = 0; index < maxFrameIndex; index++) {
063            StackTraceElement stackTraceElement = stackTraceElements[index];
064            stackTraceBuilder.add(stackTraceElement.getClassName()
065                                  + "."
066                                  + stackTraceElement.getMethodName()
067                                  + "("
068                                  + stackTraceElement.getFileName()
069                                  + ":"
070                                  + stackTraceElement.getLineNumber()
071                                  + ")");
072        }
073        if (framesInCommon > 0) {
074            stackTraceBuilder.add("... " + framesInCommon + " more");
075        }
076
077        JsonUtils.add(topLevelObjectBuilder, STACK_TRACE_PROPERTY, stackTraceBuilder);
078
079        arrayBuilder.add(topLevelObjectBuilder);
080        Throwable cause = throwable.getCause();
081        if (cause != null) {
082            addThrowable(arrayBuilder, cause, throwable);
083        }
084
085    }
086}
