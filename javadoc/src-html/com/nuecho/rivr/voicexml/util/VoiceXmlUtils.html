088    private static final Set<String> RESERVED = new HashSet<String>(asList(WORDS));
089
090    /**
091     * Checks if a name is a valid VoiceXML identifier as per VoiceXML 2.0
092     * specification (section 5.1): <blockquote>The variable naming convention
093     * is as in ECMAScript, but names beginning with the underscore character
094     * ("_") and names ending with a dollar sign ("$") are reserved for internal
095     * use. VoiceXML variables, including form item variables, must not contain
096     * ECMAScript reserved words. They must also follow ECMAScript rules for
097     * referential correctness. For example, variable names must be unique and
098     * their declaration must not include a dot - "var x.y" is an illegal
099     * declaration in ECMAScript. Variable names which violate naming
100     * conventions or ECMAScript rules cause an 'error.semantic' event to be
101     * thrown. </blockquote>
102     * 
103     * @param name the name to be tested
104     * @return <code>true</code> if and only if the name is a valid VoiceXML
105     *         identifier.
106     */
107
108    public static boolean isValidIdentifierName(String name) {
109        Assert.notNull(name, "name");
110        return ALLOWED.matcher(name).matches()
111               && !RESERVED.contains(name)
112               && !name.startsWith("_")
113               && !name.startsWith("$");
114    }
