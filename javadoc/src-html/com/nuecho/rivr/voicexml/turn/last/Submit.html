021/**
022 * Terminates a {@link VoiceXmlDialogue} with a submit, sending information to
023 * the server and leaving control to the document returned by the request.
024 * Submit can be performed with a <code>GET</code> or a <code>POST</code>.
025 * 
026 * @author Nu Echo Inc.
027 * @see VoiceXmlDialogue
028 * @see LastTurn
029 * @see SubmitMethod
030 * @see <a
031 *      href="https://www.w3.org/TR/voicexml20/#dml5.3.8">https://www.w3.org/TR/voicexml20/#dml5.3.8</a>
032 */
033public class Submit extends VoiceXmlLastTurn {
034
035    private static final String METHOD_PROPERTY = "method";
036    private static final String MEDIA_ENCODING_TYPE_PROPERTY = "mediaEncodingType";
037    private static final String FETCH_CONFIGURATION_PROPERTY = "fetchConfiguration";
038    private static final String URI_PROPERTY = "uri";
039    private static final String VARIABLES_PROPERTY = "variables";
040
041    private final VariableList mVariables;
042    private final String mUri;
043
044    private SubmitMethod mMethod = SubmitMethod.get;
045    private String mMediaEncodingType;
046
047    private DocumentFetchConfiguration mFetchConfiguration;
048
049    public Submit(String name, String uri, VariableList variables) {
050        super(name);
051        Assert.notNull(uri, "uri");
052        mVariables = variables;
053        mUri = uri;
054    }
055
056    public DocumentFetchConfiguration getFetchConfiguration() {
057        return mFetchConfiguration;
058    }
059
060    public void setFetchConfiguration(DocumentFetchConfiguration fetchConfiguration) {
061        mFetchConfiguration = fetchConfiguration;
062    }
063
064    public void setMethod(SubmitMethod method) {
065        mMethod = method;
066    }
067
068    public SubmitMethod getMethod() {
069        return mMethod;
070    }
071
072    public String getMediaEncodingType() {
073        return mMediaEncodingType;
074    }
075
076    public void setMediaEncodingType(String mediaEncodingType) {
077        mMediaEncodingType = mediaEncodingType;
078    }
079
080    public VariableList getVariables() {
081        return mVariables;
082    }
083
084    public String getUri() {
085        return mUri;
086    }
087
088    @Override
089    protected void addTurnProperties(JsonObjectBuilder builder) {
090        JsonUtils.add(builder, VARIABLES_PROPERTY, mVariables);
091        JsonUtils.add(builder, URI_PROPERTY, mUri);
092        JsonUtils.add(builder, FETCH_CONFIGURATION_PROPERTY, mFetchConfiguration);
093        JsonUtils.add(builder, MEDIA_ENCODING_TYPE_PROPERTY, mMediaEncodingType);
094        if (mMethod != null) {
095            JsonUtils.add(builder, METHOD_PROPERTY, mMethod.name());
096        }
097    }
098
099    @Override
100    protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
101            throws VoiceXmlDocumentRenderingException {
102        Element blockElement = DomUtils.appendNewElement(formElement, BLOCK_ELEMENT);
103        Element submitElement = document.createElement(SUBMIT_ELEMENT);
104        VoiceXmlDomUtil.setAttribute(submitElement, VoiceXmlDomUtil.NEXT_ATTRIBUTE, mUri);
105        addNamelist(blockElement, submitElement, mVariables);
106        VoiceXmlDomUtil.applyDocumentFetchConfiguration(submitElement, mFetchConfiguration);
107        blockElement.appendChild(submitElement);
108    }
109
110    @Override
111    public int hashCode() {
112        final int prime = 31;
113        int result = super.hashCode();
114        result = prime * result + ((mFetchConfiguration == null) ? 0 : mFetchConfiguration.hashCode());
115        result = prime * result + ((mMediaEncodingType == null) ? 0 : mMediaEncodingType.hashCode());
116        result = prime * result + ((mMethod == null) ? 0 : mMethod.hashCode());
117        result = prime * result + ((mUri == null) ? 0 : mUri.hashCode());
118        result = prime * result + ((mVariables == null) ? 0 : mVariables.hashCode());
119        return result;
120    }
121
122    @Override
123    public boolean equals(Object obj) {
124        if (this == obj) return true;
125        if (!super.equals(obj)) return false;
126        if (getClass() != obj.getClass()) return false;
127        Submit other = (Submit) obj;
128        if (mFetchConfiguration == null) {
129            if (other.mFetchConfiguration != null) return false;
130        } else if (!mFetchConfiguration.equals(other.mFetchConfiguration)) return false;
131        if (mMediaEncodingType == null) {
132            if (other.mMediaEncodingType != null) return false;
133        } else if (!mMediaEncodingType.equals(other.mMediaEncodingType)) return false;
134        if (mMethod != other.mMethod) return false;
135        if (mUri == null) {
136            if (other.mUri != null) return false;
137        } else if (!mUri.equals(other.mUri)) return false;
138        if (mVariables == null) {
139            if (other.mVariables != null) return false;
140        } else if (!mVariables.equals(other.mVariables)) return false;
141        return true;
142    }
143
144}
