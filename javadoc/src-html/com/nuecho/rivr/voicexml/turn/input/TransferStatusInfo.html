004package com.nuecho.rivr.voicexml.turn.input;
005
006import javax.json.*;
007
008import com.nuecho.rivr.core.util.*;
009import com.nuecho.rivr.voicexml.turn.output.*;
010import com.nuecho.rivr.voicexml.util.json.*;
011
012/**
013 * {@link TransferStatusInfo} is the result of a {@link Transfer} and is
014 * composed of a {@link TransferStatus} and the transfer {@link Duration}.
015 * 
016 * @author Nu Echo Inc.
017 * @see TransferStatus
018 */
019public final class TransferStatusInfo implements JsonSerializable {
020
021    private static final String DURATION_PROPERTY = "duration";
022    private static final String STATUS_PROPERTY = "status";
023    private final TransferStatus mStatus;
024    private final Duration mDuration;
025
026    public TransferStatusInfo(TransferStatus status, Duration duration) {
027        Assert.notNull(status, STATUS_PROPERTY);
028        Assert.notNull(duration, DURATION_PROPERTY);
029        mStatus = status;
030        mDuration = duration;
031    }
032
033    public TransferStatus getStatus() {
034        return mStatus;
035    }
036
037    public Duration getDuration() {
038        return mDuration;
039    }
040
041    @Override
042    public String toString() {
043        return asJson().toString();
044    }
045
046    @Override
047    public JsonValue asJson() {
048        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
049        JsonUtils.add(builder, STATUS_PROPERTY, mStatus.asJson());
050        builder.add(DURATION_PROPERTY, mDuration.getMilliseconds());
051        return builder.build();
052    }
053
054    @Override
055    public int hashCode() {
056        final int prime = 31;
057        int result = 1;
058        result = prime * result + ((mDuration == null) ? 0 : mDuration.hashCode());
059        result = prime * result + ((mStatus == null) ? 0 : mStatus.hashCode());
060        return result;
061    }
062
063    @Override
064    public boolean equals(Object obj) {
065        if (this == obj) return true;
066        if (obj == null) return false;
067        if (getClass() != obj.getClass()) return false;
068        TransferStatusInfo other = (TransferStatusInfo) obj;
069        if (mDuration == null) {
070            if (other.mDuration != null) return false;
071        } else if (!mDuration.equals(other.mDuration)) return false;
072        if (mStatus == null) {
073            if (other.mStatus != null) return false;
074        } else if (!mStatus.equals(other.mStatus)) return false;
075        return true;
076    }
077
078}
