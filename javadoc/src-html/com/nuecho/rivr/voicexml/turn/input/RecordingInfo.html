004package com.nuecho.rivr.voicexml.turn.input;
005
006import javax.json.*;
007
008import com.nuecho.rivr.core.util.*;
009import com.nuecho.rivr.voicexml.servlet.*;
010import com.nuecho.rivr.voicexml.turn.output.*;
011import com.nuecho.rivr.voicexml.turn.output.Interaction.FinalRecordingWindow;
012import com.nuecho.rivr.voicexml.util.json.*;
013
014/**
015 * {@link RecordingInfo} contains the recorded file, its duration and other meta
016 * information produced by an {@link Interaction} with a
017 * {@link FinalRecordingWindow}.
018 * 
019 * @author Nu Echo Inc.
020 * @see Interaction
021 * @see FinalRecordingWindow
022 * @see Recording
023 * @see FileUpload
024 */
025public final class RecordingInfo implements JsonSerializable {
026
027    private static final String DURATION_PROPERTY = "duration";
028    private static final String MAX_TIME_PROPERTY = "maxTime";
029    private static final String DTMF_TERM_PROPERTY = "dtmfTerm";
030
031    private final FileUpload mFile;
032    private final Duration mDuration;
033    private final boolean mMaxTime;
034    private final String mDtmfTerm;
035
036    public RecordingInfo(FileUpload file, Duration duration, boolean maxTime, String dtmfTerm) {
037        mFile = file;
038        mDuration = duration;
039        mMaxTime = maxTime;
040        mDtmfTerm = dtmfTerm;
041    }
042
043    public FileUpload getFile() {
044        return mFile;
045    }
046
047    public boolean isMaxTime() {
048        return mMaxTime;
049    }
050
051    public String getDtmfTerm() {
052        return mDtmfTerm;
053    }
054
055    public Duration getDuration() {
056        return mDuration;
057    }
058
059    @Override
060    public String toString() {
061        return asJson().toString();
062    }
063
064    @Override
065    public JsonValue asJson() {
066        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
067        JsonUtils.add(builder, DTMF_TERM_PROPERTY, mDtmfTerm);
068        builder.add(MAX_TIME_PROPERTY, mMaxTime);
069        builder.add(DURATION_PROPERTY, mDuration.getMilliseconds());
070        JsonUtils.add(builder, "file", mFile);
071        return builder.build();
072    }
073
074    @Override
075    public int hashCode() {
076        final int prime = 31;
077        int result = 1;
078        result = prime * result + ((mDtmfTerm == null) ? 0 : mDtmfTerm.hashCode());
079        result = prime * result + ((mDuration == null) ? 0 : mDuration.hashCode());
080        result = prime * result + ((mFile == null) ? 0 : mFile.hashCode());
081        result = prime * result + (mMaxTime ? 1231 : 1237);
082        return result;
083    }
084
085    @Override
086    public boolean equals(Object obj) {
087        if (this == obj) return true;
088        if (obj == null) return false;
089        if (getClass() != obj.getClass()) return false;
090        RecordingInfo other = (RecordingInfo) obj;
091        if (mDtmfTerm == null) {
092            if (other.mDtmfTerm != null) return false;
093        } else if (!mDtmfTerm.equals(other.mDtmfTerm)) return false;
094        if (mDuration == null) {
095            if (other.mDuration != null) return false;
096        } else if (!mDuration.equals(other.mDuration)) return false;
097        if (mFile == null) {
098            if (other.mFile != null) return false;
099        } else if (!mFile.equals(other.mFile)) return false;
100        if (mMaxTime != other.mMaxTime) return false;
101        return true;
102    }
103
104}
