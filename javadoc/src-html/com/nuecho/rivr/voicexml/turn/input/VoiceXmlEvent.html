005package com.nuecho.rivr.voicexml.turn.input;
006
007import java.util.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.core.util.*;
012import com.nuecho.rivr.voicexml.util.json.*;
013
014/**
015 * A VoiceXml event thrown by the platform composed of a name and a message.
016 * 
017 * @author Nu Echo Inc.
018 * @see <a
019 *      href="https://www.w3.org/TR/voicexml20/#dml5.2.6">https://www.w3.org/TR/voicexml20/#dml5.2.6</a>
020 */
021public final class VoiceXmlEvent implements JsonSerializable {
022
023    private static final String MESSAGE_PROPERT = "message";
024    private static final String NAME_PROPERTY = "name";
025
026    private final String mName;
027    private final String mMessage;
028
029    public static final String CANCEL = "cancel";
030    public static final String CONNECTION_DISCONNECT = "connection.disconnect";
031    public static final String CONNECTION_DISCONNECT_HANGUP = "connection.disconnect.hangup";
032    public static final String CONNECTION_DISCONNECT_TRANSFER = "connection.disconnect.transfer";
033    public static final String EXIT = "exit";
034    public static final String HELP = "help";
035    public static final String NO_INPUT = "noinput";
036    public static final String NO_MATCH = "nomatch";
037    public static final String MAX_SPEECH_TIMEOUT = "maxspeechtimeout";
038    public static final String ERROR = "error";
039    public static final String ERROR_BAD_FETCH = "error.badfetch";
040    public static final String ERROR_BAD_FETCH_HTTP = "error.badfetch.http";
041    public static final String ERROR_BAD_FETCH_PROTOCOL = "error.badfetch.protocol";
042    public static final String ERROR_SEMANTIC = "error.semantic";
043    public static final String ERROR_NO_AUTHORIZATION = "error.noauthorization";
044    public static final String ERROR_NO_RESOURCE = "error.noresource";
045    public static final String ERROR_UNSUPPORTED_BUILTIN = "error.unsupported.builtin";
046    public static final String ERROR_UNSUPPORTED_FORMAT = "error.unsupported.format";
047    public static final String ERROR_UNSUPPORTED_LANGUAGE = "error.unsupported.language";
048    public static final String ERROR_UNSUPPORTED_OBJECT_NAME = "error.unsupported.objectname";
049    public static final String ERROR_UNSUPPORTED = "error.unsupported";
050
051    public static final String ERROR_CONNECTION = "error.connection";
052    public static final String ERROR_CONNECTION_NO_AUTHORIZATION = "error.connection.noauthorization";
053    public static final String ERROR_CONNECTION_BAD_DESTINATION = "error.connection.baddestination";
054    public static final String ERROR_CONNECTION_NO_ROUTE = "error.connection.noroute";
055    public static final String ERROR_CONNECTION_NO_RESOURCE = "error.connection.noresource";
056    public static final String ERROR_CONNECTION_PROTOCOL = "error.connection.protocol";
057    public static final String ERROR_UNSUPPORTED_TRANSFER_BLIND = "error.unsupported.transfer.blind";
058    public static final String ERROR_UNSUPPORTED_TRANSFER_BRIDGE = "error.unsupported.transfer.bridge";
059    public static final String ERROR_UNSUPPORTED_TRANSFER_CONSULTATION = "error.unsupported.transfer.consultation";
060    public static final String ERROR_UNSUPPORTED_URI = "error.unsupported.uri";
061
062    public static boolean hasEvent(String prefix, Collection<? extends VoiceXmlEvent> events) {
063        for (VoiceXmlEvent voiceXmlEvent : events) {
064            if (voiceXmlEvent.isSubtypeOf(prefix)) return true;
065        }
066        return false;
067    }
068
069    public static VoiceXmlEvent getEvent(String prefix, Collection<? extends VoiceXmlEvent> events) {
070        for (VoiceXmlEvent voiceXmlEvent : events) {
071            if (voiceXmlEvent.isSubtypeOf(prefix)) return voiceXmlEvent;
072        }
073        return null;
074    }
075
076    public VoiceXmlEvent(String name, String message) {
077        Assert.notEmpty(name, NAME_PROPERTY);
078        mName = name;
079        mMessage = message;
080    }
081
082    public String getName() {
083        return mName;
084    }
085
086    public String getMessage() {
087        return mMessage;
088    }
089
090    public boolean isSubtypeOf(String prefix) {
091        Assert.notNull(prefix, "prefix");
092        return mName.equals(prefix) || mName.startsWith(prefix + ".");
093    }
094
095    @Override
096    public String toString() {
097        return asJson().toString();
098    }
099
100    @Override
101    public JsonValue asJson() {
102        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
103        JsonUtils.add(builder, NAME_PROPERTY, mName);
104        JsonUtils.add(builder, MESSAGE_PROPERT, mMessage);
105        return builder.build();
106    }
107
108    @Override
109    public int hashCode() {
110        final int prime = 31;
111        int result = 1;
112        result = prime * result + ((mMessage == null) ? 0 : mMessage.hashCode());
113        result = prime * result + ((mName == null) ? 0 : mName.hashCode());
114        return result;
115    }
116
117    @Override
118    public boolean equals(Object obj) {
119        if (this == obj) return true;
120        if (obj == null) return false;
121        if (getClass() != obj.getClass()) return false;
122        VoiceXmlEvent other = (VoiceXmlEvent) obj;
123        if (mMessage == null) {
124            if (other.mMessage != null) return false;
125        } else if (!mMessage.equals(other.mMessage)) return false;
126        if (mName == null) {
127            if (other.mName != null) return false;
128        } else if (!mName.equals(other.mName)) return false;
129        return true;
130    }
131
132}
