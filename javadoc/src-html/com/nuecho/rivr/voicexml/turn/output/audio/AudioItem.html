005package com.nuecho.rivr.voicexml.turn.output.audio;
006
007import javax.json.*;
008
009import com.nuecho.rivr.voicexml.util.json.*;
010
011/**
012 * This abstract class is the superclass of all classes representing an audio
013 * element in the dialogue. It is not intended to be subclassed.
014 * 
015 * @author Nu Echo Inc.
016 * @see Mark
017 * @see Pause
018 * @see AudioFile
019 * @see SpeechSynthesis
020 * @see <a
021 *      href="https://www.w3.org/TR/voicexml20/#dml4.1">https://www.w3.org/TR/voicexml20/#dml4.1</a>
022 */
023public abstract class AudioItem implements JsonSerializable {
024    public static final String TYPE_PROPERTY = "type";
025
026    public abstract String getElementType();
027
028    protected abstract void addJsonProperties(JsonObjectBuilder builder);
029
030    @Override
031    public final String toString() {
032        return asJson().toString();
033    }
034
035    @Override
036    public final JsonValue asJson() {
037        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
038        JsonUtils.add(builder, TYPE_PROPERTY, getElementType());
039        addJsonProperties(builder);
040        return builder.build();
041    }
042}
