004package com.nuecho.rivr.voicexml.turn.output.audio;
005
006import javax.json.*;
007
008import com.nuecho.rivr.core.util.*;
009import com.nuecho.rivr.voicexml.turn.input.*;
010import com.nuecho.rivr.voicexml.util.json.*;
011
012/**
013 * A {@link Mark} represents a marker in an {@link AudioItem} sequence which can
014 * be used to determine when a user started to speak.
015 * 
016 * @author Nu Echo Inc.
017 * @see MarkInfo
018 */
019public final class Mark extends AudioItem {
020    private static final String MARK_PROPERTY = "mark";
021    private static final String MARK_ELEMENT_TYPE = "mark";
022    private final String mName;
023
024    /**
025     * @param name The name of the mark. Not null.
026     */
027    public Mark(String name) {
028        Assert.notNull(name, "name");
029        mName = name;
030    }
031
032    @Override
033    public String getElementType() {
034        return MARK_ELEMENT_TYPE;
035    }
036
037    public String getName() {
038        return mName;
039    }
040
041    @Override
042    protected void addJsonProperties(JsonObjectBuilder builder) {
043        JsonUtils.add(builder, MARK_PROPERTY, mName);
044    }
045
046    @Override
047    public int hashCode() {
048        final int prime = 31;
049        int result = 1;
050        result = prime * result + (mName == null ? 0 : mName.hashCode());
051        return result;
052    }
053
054    @Override
055    public boolean equals(Object obj) {
056        if (this == obj) return true;
057        if (obj == null) return false;
058        if (getClass() != obj.getClass()) return false;
059        Mark other = (Mark) obj;
060        if (mName == null) {
061            if (other.mName != null) return false;
062        } else if (!mName.equals(other.mName)) return false;
063        return true;
064    }
065}
