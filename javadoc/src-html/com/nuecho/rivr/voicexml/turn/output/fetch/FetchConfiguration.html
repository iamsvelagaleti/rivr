005package com.nuecho.rivr.voicexml.turn.output.fetch;
006
007import javax.json.*;
008
009import com.nuecho.rivr.core.util.*;
010import com.nuecho.rivr.voicexml.turn.output.*;
011import com.nuecho.rivr.voicexml.turn.output.audio.*;
012import com.nuecho.rivr.voicexml.turn.output.grammar.*;
013import com.nuecho.rivr.voicexml.util.json.*;
014
015/**
016 * Fetch properties for a resource:
017 * <ul>
018 * <li>maxage</li>
019 * <li>maxstale</li>
020 * <li>timeout</li>
021 * <li>fetchhint</li>
022 * </ul>
023 * 
024 * @see AudioFile#setFetchConfiguration(FetchConfiguration)
025 * @see GrammarReference#setFetchConfiguration(FetchConfiguration)
026 * @see ObjectCall#setFetchConfiguration(FetchConfiguration)
027 * @author Nu Echo Inc.
028 */
029public class FetchConfiguration implements JsonSerializable {
030
031    private static final String MAX_AGE_PROPERTY = "maxAge";
032    private static final String MAX_STALE_PROPERTY = "maxStale";
033    private static final String TIME_OUT_PROPERTY = "timeOut";
034    private static final String FETCH_HINT_PROPERTY = "fetchHint";
035
036    private Duration mTimeOut;
037    private Duration mMaxAge;
038    private Duration mMaxStale;
039    private FetchHint mFetchHint;
040
041    public Duration getTimeOut() {
042        return mTimeOut;
043    }
044
045    public Duration getMaxAge() {
046        return mMaxAge;
047    }
048
049    public Duration getMaxStale() {
050        return mMaxStale;
051    }
052
053    public FetchHint getFetchHint() {
054        return mFetchHint;
055    }
056
057    public void setTimeOut(Duration timeOut) {
058        mTimeOut = timeOut;
059    }
060
061    public void setMaxAge(Duration maxAge) {
062        mMaxAge = maxAge;
063    }
064
065    public void setMaxStale(Duration maxStale) {
066        mMaxStale = maxStale;
067    }
068
069    public void setFetchHint(FetchHint fetchHint) {
070        mFetchHint = fetchHint;
071    }
072
073    @Override
074    public String toString() {
075        return asJson().toString();
076    }
077
078    @Override
079    public JsonValue asJson() {
080        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
081        JsonUtils.addDurationProperty(builder, MAX_AGE_PROPERTY, mMaxAge);
082        JsonUtils.addDurationProperty(builder, MAX_STALE_PROPERTY, mMaxStale);
083        JsonUtils.addDurationProperty(builder, TIME_OUT_PROPERTY, mTimeOut);
084
085        if (mFetchHint == null) {
086            builder.addNull(FETCH_HINT_PROPERTY);
087        } else {
088            JsonUtils.add(builder, FETCH_HINT_PROPERTY, mFetchHint.name());
089        }
090
091        return builder.build();
092    }
093
094    @Override
095    public int hashCode() {
096        final int prime = 31;
097        int result = 1;
098        result = prime * result + (mFetchHint == null ? 0 : mFetchHint.hashCode());
099        result = prime * result + (mMaxAge == null ? 0 : mMaxAge.hashCode());
100        result = prime * result + (mMaxStale == null ? 0 : mMaxStale.hashCode());
101        result = prime * result + (mTimeOut == null ? 0 : mTimeOut.hashCode());
102        return result;
103    }
104
105    @Override
106    public boolean equals(Object obj) {
107        if (this == obj) return true;
108        if (obj == null) return false;
109        if (getClass() != obj.getClass()) return false;
110        FetchConfiguration other = (FetchConfiguration) obj;
111        if (mFetchHint != other.mFetchHint) return false;
112        if (mMaxAge == null) {
113            if (other.mMaxAge != null) return false;
114        } else if (!mMaxAge.equals(other.mMaxAge)) return false;
115        if (mMaxStale == null) {
116            if (other.mMaxStale != null) return false;
117        } else if (!mMaxStale.equals(other.mMaxStale)) return false;
118        if (mTimeOut == null) {
119            if (other.mTimeOut != null) return false;
120        } else if (!mTimeOut.equals(other.mTimeOut)) return false;
121        return true;
122    }
123
124}
