005package com.nuecho.rivr.voicexml.turn.output;
006
007import java.util.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.core.util.*;
012import com.nuecho.rivr.voicexml.turn.output.grammar.*;
013import com.nuecho.rivr.voicexml.util.json.*;
014
015/**
016 * A {@link SpeechRecognition} represents the speech part of an interaction
017 * phase recognition configuration. It is composed of a list of speech
018 * {@link GrammarItem} and optional properties (complete timeout, incomplete
019 * timeout, max speech timeout, max nbest, confidence level, speed versus
020 * accuracy and sensitivity).
021 * 
022 * @author Nu Echo Inc.
023 */
024public final class SpeechRecognition extends Recognition {
025
026    private static final String SPEED_VERSUS_ACCURACY_PROPERTY = "speedVersusAccuracy";
027    private static final String SENSITIVITY_PROPERTY = "sensitivity";
028    private static final String MAX_N_BEST_PROPERTY = "maxNBest";
029    private static final String CONFIDENCE_LEVEL_PROPERTY = "confidenceLevel";
030    private static final String MAX_SPEECH_TIMEOUT_PROPERTY = "maxSpeechTimeout";
031    private static final String INCOMPLETE_TIMEOUT_PROPERTY = "incompleteTimeout";
032    private static final String COMPLETE_TIMEOUT_PROPERTY = "completeTimeout";
033    private Duration mCompleteTimeout;
034    private Duration mIncompleteTimeout;
035    private Duration mMaxSpeechTimeout;
036    private Integer mMaxNBest;
037
038    private Double mConfidenceLevel;
039    private Double mSpeedVersusAccuracy;
040    private Double mSensitivity;
041
042    /**
043     * @param grammarItems The list of speech {@link GrammarItem}. Not null.
044     */
045    public SpeechRecognition(GrammarItem... grammarItems) {
046        super(grammarItems);
047    }
048
049    /**
050     * @param grammarItems The list of speech {@link GrammarItem}. Not null.
051     */
052    public SpeechRecognition(List<GrammarItem> grammarItems) {
053        super(grammarItems);
054    }
055
056    /**
057     * @param completeTimeout The required length of silence following user
058     *            speech before the speech recognizer finalizes a complete
059     *            result. <code>null</code> to use the VoiceXML platform default
060     * @see <a
061     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
062     */
063    public void setCompleteTimeout(Duration completeTimeout) {
064        mCompleteTimeout = completeTimeout;
065    }
066
067    /**
068     * @param incompleteTimeout The required length of silence following user
069     *            speech after which a recognizer finalizes an incomplete
070     *            result. <code>null</code> to use the VoiceXML platform default
071     * @see <a
072     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
073     */
074    public void setIncompleteTimeout(Duration incompleteTimeout) {
075        mIncompleteTimeout = incompleteTimeout;
076    }
077
078    /**
079     * @param maxSpeechTimeout The maximum duration of user speech.
080     *            <code>null</code> to use the VoiceXML platform default
081     * @see <a
082     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
083     */
084    public void setMaxSpeechTimeout(Duration maxSpeechTimeout) {
085        mMaxSpeechTimeout = maxSpeechTimeout;
086    }
087
088    /**
089     * @param maxNBest The maximum size of the recognition result. Must be a
090     *            positive integer. <code>null</code> to use the VoiceXML
091     *            platform default
092     * @see <a
093     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
094     */
095    public void setMaxNBest(Integer maxNBest) {
096        if (maxNBest != null) {
097            Assert.positive(maxNBest, "maxNBest");
098        }
099
100        mMaxNBest = maxNBest;
101    }
102
103    /**
104     * @param confidenceLevel The speech recognition confidence level. Results
105     *            with confidence below this value will be rejected (nomatch).
106     *            Value must be between 0.0 and 1.0.<code>null</code> to use the
107     *            VoiceXML platform default.
108     * @see <a
109     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
110     */
111    public void setConfidenceLevel(Double confidenceLevel) {
112        if (confidenceLevel != null) {
113            Assert.between(0, confidenceLevel, 1.0);
114        }
115
116        mConfidenceLevel = confidenceLevel;
117    }
118
119    /**
120     * @param speedVersusAccuracy A hint specifying the desired balance between
121     *            speed versus accuracy. A value of 0.0 means fastest
122     *            recognition. A value of 1.0 means best accuracy.
123     *            <code>null</code> to use the VoiceXML platform default
124     * @see <a
125     *      href="https://www.w3.org/TR/voicexml20/#dml6.3.2">https://www.w3.org/TR/voicexml20/#dml6.3.2</a>
126     */
127    public void setSpeedVersusAccuracy(Double speedVersusAccuracy) {
128        if (speedVersusAccuracy != null) {
129            Assert.between(0, speedVersusAccuracy, 1.0);
130        }
131
132        mSpeedVersusAccuracy = speedVersusAccuracy;
133    }
134
135    /**
136     * @param sensitivity Set the sensitivity level. A value of 1.0 means that
137     *            it is highly sensitive to quiet input. A value of 0.0 means it
138     *            is least sensitive to noise. <code>null</code> to use the
139     *            VoiceXML platform default.
140     */
141    public void setSensitivity(Double sensitivity) {
142        if (sensitivity != null) {
143            Assert.between(0, sensitivity, 1.0);
144        }
145
146        mSensitivity = sensitivity;
147    }
148
149    public Duration getCompleteTimeout() {
150        return mCompleteTimeout;
151    }
152
153    public Duration getIncompleteTimeout() {
154        return mIncompleteTimeout;
155    }
156
157    public Duration getMaxSpeechTimeout() {
158        return mMaxSpeechTimeout;
159    }
160
161    public Integer getMaxNBest() {
162        return mMaxNBest;
163    }
164
165    public Double getConfidenceLevel() {
166        return mConfidenceLevel;
167    }
168
169    public Double getSpeedVersusAccuracy() {
170        return mSpeedVersusAccuracy;
171    }
172
173    public Double getSensitivity() {
174        return mSensitivity;
175    }
176
177    @Override
178    protected void addJsonProperties(JsonObjectBuilder builder) {
179        JsonUtils.addDurationProperty(builder, COMPLETE_TIMEOUT_PROPERTY, mCompleteTimeout);
180        JsonUtils.addDurationProperty(builder, INCOMPLETE_TIMEOUT_PROPERTY, mIncompleteTimeout);
181        JsonUtils.addDurationProperty(builder, MAX_SPEECH_TIMEOUT_PROPERTY, mMaxSpeechTimeout);
182        JsonUtils.addNumberProperty(builder, CONFIDENCE_LEVEL_PROPERTY, mConfidenceLevel);
183        JsonUtils.addNumberProperty(builder, MAX_N_BEST_PROPERTY, mMaxNBest);
184        JsonUtils.addNumberProperty(builder, SENSITIVITY_PROPERTY, mSensitivity);
185        JsonUtils.addNumberProperty(builder, SPEED_VERSUS_ACCURACY_PROPERTY, mSpeedVersusAccuracy);
186    }
187
188    @Override
189    public int hashCode() {
190        final int prime = 31;
191        int result = super.hashCode();
192        result = prime * result + (mCompleteTimeout == null ? 0 : mCompleteTimeout.hashCode());
193        result = prime * result + (mConfidenceLevel == null ? 0 : mConfidenceLevel.hashCode());
194        result = prime * result + (mIncompleteTimeout == null ? 0 : mIncompleteTimeout.hashCode());
195        result = prime * result + (mMaxNBest == null ? 0 : mMaxNBest.hashCode());
196        result = prime * result + (mMaxSpeechTimeout == null ? 0 : mMaxSpeechTimeout.hashCode());
197        result = prime * result + (mSensitivity == null ? 0 : mSensitivity.hashCode());
198        result = prime * result + (mSpeedVersusAccuracy == null ? 0 : mSpeedVersusAccuracy.hashCode());
199        return result;
200    }
201
202    @Override
203    public boolean equals(Object obj) {
204        if (this == obj) return true;
205        if (!super.equals(obj)) return false;
206        if (getClass() != obj.getClass()) return false;
207        SpeechRecognition other = (SpeechRecognition) obj;
208        if (mCompleteTimeout == null) {
209            if (other.mCompleteTimeout != null) return false;
210        } else if (!mCompleteTimeout.equals(other.mCompleteTimeout)) return false;
211        if (mConfidenceLevel == null) {
212            if (other.mConfidenceLevel != null) return false;
213        } else if (!mConfidenceLevel.equals(other.mConfidenceLevel)) return false;
214        if (mIncompleteTimeout == null) {
215            if (other.mIncompleteTimeout != null) return false;
216        } else if (!mIncompleteTimeout.equals(other.mIncompleteTimeout)) return false;
217        if (mMaxNBest == null) {
218            if (other.mMaxNBest != null) return false;
219        } else if (!mMaxNBest.equals(other.mMaxNBest)) return false;
220        if (mMaxSpeechTimeout == null) {
221            if (other.mMaxSpeechTimeout != null) return false;
222        } else if (!mMaxSpeechTimeout.equals(other.mMaxSpeechTimeout)) return false;
223        if (mSensitivity == null) {
224            if (other.mSensitivity != null) return false;
225        } else if (!mSensitivity.equals(other.mSensitivity)) return false;
226        if (mSpeedVersusAccuracy == null) {
227            if (other.mSpeedVersusAccuracy != null) return false;
228        } else if (!mSpeedVersusAccuracy.equals(other.mSpeedVersusAccuracy)) return false;
229        return true;
230    }
231
232    public SpeechRecognition copy() {
233        SpeechRecognition copy = new SpeechRecognition(getGrammarItems());
234        copy.mCompleteTimeout = mCompleteTimeout;
235        copy.mMaxNBest = mMaxNBest;
236        copy.mIncompleteTimeout = mIncompleteTimeout;
237        copy.mMaxSpeechTimeout = mMaxSpeechTimeout;
238        copy.mConfidenceLevel = mConfidenceLevel;
239        copy.mSensitivity = mSensitivity;
240        copy.mSpeedVersusAccuracy = mSpeedVersusAccuracy;
241        copyPropertiesTo(copy);
242        return copy;
243    }
244}
