004package com.nuecho.rivr.voicexml.turn.output.audio;
005
006import javax.json.*;
007
008import org.w3c.dom.*;
009
010import com.nuecho.rivr.core.util.*;
011import com.nuecho.rivr.voicexml.util.json.*;
012
013/**
014 * A {@link SpeechSynthesis} represents a text that will be spoken using a
015 * synthesis voice. It can be either a string of text or a SSML document.
016 * 
017 * @author Nu Echo Inc.
018 * @see <a
019 *      href="https://www.w3.org/TR/speech-synthesis/">https://www.w3.org/TR/speech-synthesis/</a>
020 */
021public final class SpeechSynthesis extends AudioItem {
022    public static final String SPEECH_SYNTHESIS_ELEMENT_TYPE = "speechSynthesis";
023    private static final String TEXT_PROPERTY = "text";
024    private static final String SSML_PROPERTY = "ssml";
025
026    private final String mText;
027    private final DocumentFragment mDocumentFragment;
028
029    /**
030     * @param text The synthesis text. Not empty.
031     */
032    public SpeechSynthesis(String text) {
033        Assert.notEmpty(text, "text");
034        mText = text;
035        mDocumentFragment = null;
036    }
037
038    /**
039     * @param documentFragment The SSML document. Not null.
040     */
041    public SpeechSynthesis(DocumentFragment documentFragment) {
042        Assert.notNull(documentFragment, "documentFragment");
043        mDocumentFragment = documentFragment;
044        mText = null;
045    }
046
047    @Override
048    public String getElementType() {
049        return SPEECH_SYNTHESIS_ELEMENT_TYPE;
050    }
051
052    public String getText() {
053        return mText;
054    }
055
056    public DocumentFragment getDocumentFragment() {
057        return mDocumentFragment;
058    }
059
060    public boolean isSsml() {
061        return mDocumentFragment != null;
062    }
063
064    @Override
065    protected void addJsonProperties(JsonObjectBuilder builder) {
066        if (isSsml()) {
067            JsonUtils.addXmlNodeProperty(builder, SSML_PROPERTY, "SSML fragment", mDocumentFragment);
068        } else {
069            builder.addNull(SSML_PROPERTY);
070        }
071
072        JsonUtils.add(builder, TEXT_PROPERTY, mText);
073    }
074
075    @Override
076    public int hashCode() {
077        final int prime = 31;
078        int result = 1;
079        result = prime * result + ((mDocumentFragment == null) ? 0 : mDocumentFragment.hashCode());
080        result = prime * result + ((mText == null) ? 0 : mText.hashCode());
081        return result;
082    }
083
084    @Override
085    public boolean equals(Object obj) {
086        if (this == obj) return true;
087        if (obj == null) return false;
088        if (getClass() != obj.getClass()) return false;
089        SpeechSynthesis other = (SpeechSynthesis) obj;
090        if (mDocumentFragment == null) {
091            if (other.mDocumentFragment != null) return false;
092        } else if (!mDocumentFragment.equals(other.mDocumentFragment)) return false;
093        if (mText == null) {
094            if (other.mText != null) return false;
095        } else if (!mText.equals(other.mText)) return false;
096        return true;
097    }
098}
