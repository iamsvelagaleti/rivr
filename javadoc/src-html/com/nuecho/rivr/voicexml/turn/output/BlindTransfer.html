004package com.nuecho.rivr.voicexml.turn.output;
005
006/**
007 * A {@link BlindTransfer} is a {@link Transfer} where the outcome is not
008 * monitored.
009 * <p>
010 * Once the transfer begins and the interpreter disconnects from the session,
011 * the platform throws connection.disconnect.transfer.
012 * 
013 * @author Nu Echo Inc.
014 * @see <a
015 *      href="https://www.w3.org/TR/voicexml20/#dml2.3.7.1">https://www.w3.org/TR/voicexml20/#dml2.3.7.1</a>
016 */
017public class BlindTransfer extends Transfer {
018    private static final String BLIND_TRANSFER_TYPE = "blind";
019
020    /**
021     * @param name The name of this turn. Not empty.
022     * @param destination The URI of the destination (telephone, IP telephony
023     *            address). Not empty.
024     */
025    public BlindTransfer(String name, String destination) {
026        super(name, destination);
027    }
028
029    @Override
030    protected final String getTransferType() {
031        return BLIND_TRANSFER_TYPE;
032    }
033
034    /**
035     * Builder used to ease the creation of instances of {@link BlindTransfer}.
036     */
037    public static class Builder extends Transfer.Builder {
038
039        public Builder(String name) {
040            super(name);
041        }
042
043        public BlindTransfer build() {
044            BlindTransfer blindTransfer = new BlindTransfer(getName(), getDestination());
045            super.build(blindTransfer);
046            return blindTransfer;
047
048        }
049
050    }
051
052}
