005package com.nuecho.rivr.voicexml.turn.output;
006
007import javax.json.*;
008
009import com.nuecho.rivr.core.util.*;
010import com.nuecho.rivr.voicexml.turn.input.*;
011import com.nuecho.rivr.voicexml.util.json.*;
012
013/**
014 * A {@link Recording} represents the description of a recording final phase of
015 * an interaction.
016 * <p>
017 * It may have an optional {@link DtmfRecognition} to recognize DTMF input while
018 * recording. A DTMF input matching one of the {@link DtmfRecognition} grammar
019 * will terminate the recording and place the recording in a variable.
020 * 
021 * @author Nu Echo Inc.
022 * @see DtmfRecognition
023 * @see <a
024 *      href="https://www.w3.org/TR/voicexml20/#dml2.3.6">https://www.w3.org/TR/voicexml20/#dml2.3.6</a>
025 */
026public final class Recording implements JsonSerializable {
027    private static final String DTMF_TERM_PROPERTY = "dtmfTerm";
028    private static final String BEEP_PROPERTY = "beep";
029    private static final String POST_AUDIO_TO_SERVER_PROPERTY = "postAudioToServer";
030    private static final String MAXIMUM_TIME_PROPERTY = "maximumTime";
031    private static final String FINAL_SILENCE_PROPERTY = "finalSilence";
032    private static final String DTMF_TERM_RECOGNITION_PROPERTY = "dtmfTermRecognition";
033    private static final String CLIENT_SIDE_ASSIGNATION_DESTINATION_PROPERTY = "clientSideAssignationDestination";
034    private static final String TYPE_PROPERTY = "type";
035
036    private DtmfRecognition mDtmfTermRecognition;
037    private Boolean mBeep;
038    private Duration mMaximumTime;
039    private Duration mFinalSilence;
040    private String mType;
041    private String mClientSideAssignationDestination;
042    private Boolean mDtmfTerm;
043    private boolean mPostAudioToServer = true;
044
045    /**
046     * @param dtmfRecognition The {@link DtmfRecognition} used to interrupt the
047     *            recording. <code>null</code> to use the VoiceXML platform
048     *            default
049     */
050    public void setDtmfTermRecognition(DtmfRecognition dtmfRecognition) {
051        mDtmfTermRecognition = dtmfRecognition;
052    }
053
054    /**
055     * @param beep If true, a tone is emitted just prior to recording.
056     *            <code>null</code> to use the VoiceXML platform default.
057     */
058    public void setBeep(Boolean beep) {
059        mBeep = Boolean.valueOf(beep);
060    }
061
062    /**
063     * @param maximumTime The maximum duration to record. <code>null</code> to
064     *            use the VoiceXML platform default.
065     */
066    public void setMaximumTime(Duration maximumTime) {
067        mMaximumTime = maximumTime;
068    }
069
070    /**
071     * @param finalSilence The interval of silence that indicates end of speech.
072     *            <code>null</code> to use the VoiceXML platform default
073     */
074    public void setFinalSilence(Duration finalSilence) {
075        mFinalSilence = finalSilence;
076    }
077
078    /**
079     * @param type The media format of the resulting recording.
080     *            <code>null</code> to use the VoiceXML platform default.
081     * @see <a
082     *      href="https://www.w3.org/TR/voicexml20/#dmlAAudioFormats">https://www.w3.org/TR/voicexml20/#dmlAAudioFormats</a>
083     */
084    public void setType(String type) {
085        mType = type;
086    }
087
088    /**
089     * @param clientSideAssignationDestination The variable where the recording
090     *            will be stored.
091     */
092    public void setClientSideAssignationDestination(String clientSideAssignationDestination) {
093        mClientSideAssignationDestination = clientSideAssignationDestination;
094    }
095
096    /**
097     * @param dtmfTerm If true, any DTMF keypress not matched by an active
098     *            grammar will be treated as a match of an active local DTMF
099     *            grammar. <code>null</code> to use the VoiceXML platform
100     *            default
101     */
102    public void setDtmfTerm(Boolean dtmfTerm) {
103        mDtmfTerm = Boolean.valueOf(dtmfTerm);
104    }
105
106    /**
107     * @param postAudioToServer If true, recording will be posted to server and
108     *            made available in the <code>file</code> property of the
109     *            {@link RecordingInfo}.
110     */
111    public void setPostAudioToServer(boolean postAudioToServer) {
112        mPostAudioToServer = postAudioToServer;
113    }
114
115    public DtmfRecognition getDtmfTermRecognition() {
116        return mDtmfTermRecognition;
117    }
118
119    public Boolean getBeep() {
120        return mBeep;
121    }
122
123    public Duration getMaximumTime() {
124        return mMaximumTime;
125    }
126
127    public Duration getFinalSilence() {
128        return mFinalSilence;
129    }
130
131    public Boolean getDtmfTerm() {
132        return mDtmfTerm;
133    }
134
135    public String getType() {
136        return mType;
137    }
138
139    public String getClientSideAssignationDestination() {
140        return mClientSideAssignationDestination;
141    }
142
143    public boolean isPostAudioToServer() {
144        return mPostAudioToServer;
145    }
146
147    @Override
148    public JsonValue asJson() {
149        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
150        JsonUtils.addBooleanProperty(builder, BEEP_PROPERTY, mBeep);
151        JsonUtils.addBooleanProperty(builder, DTMF_TERM_PROPERTY, mDtmfTerm);
152        JsonUtils.addBooleanProperty(builder, POST_AUDIO_TO_SERVER_PROPERTY, mPostAudioToServer);
153        JsonUtils.addDurationProperty(builder, MAXIMUM_TIME_PROPERTY, mMaximumTime);
154        JsonUtils.addDurationProperty(builder, FINAL_SILENCE_PROPERTY, mFinalSilence);
155        JsonUtils.add(builder, DTMF_TERM_RECOGNITION_PROPERTY, mDtmfTermRecognition);
156        JsonUtils.add(builder, TYPE_PROPERTY, mType);
157        JsonUtils.add(builder, CLIENT_SIDE_ASSIGNATION_DESTINATION_PROPERTY, mClientSideAssignationDestination);
158        return builder.build();
159    }
160
161    @Override
162    public int hashCode() {
163        final int prime = 31;
164        int result = 1;
165        result = prime * result + ((mBeep == null) ? 0 : mBeep.hashCode());
166        result = prime
167                 * result
168                 + ((mClientSideAssignationDestination == null) ? 0 : mClientSideAssignationDestination.hashCode());
169        result = prime * result + ((mDtmfTerm == null) ? 0 : mDtmfTerm.hashCode());
170        result = prime * result + ((mDtmfTermRecognition == null) ? 0 : mDtmfTermRecognition.hashCode());
171        result = prime * result + ((mFinalSilence == null) ? 0 : mFinalSilence.hashCode());
172        result = prime * result + ((mMaximumTime == null) ? 0 : mMaximumTime.hashCode());
173        result = prime * result + (mPostAudioToServer ? 1231 : 1237);
174        result = prime * result + ((mType == null) ? 0 : mType.hashCode());
175        return result;
176    }
177
178    @Override
179    public boolean equals(Object obj) {
180        if (this == obj) return true;
181        if (obj == null) return false;
182        if (getClass() != obj.getClass()) return false;
183        Recording other = (Recording) obj;
184        if (mBeep == null) {
185            if (other.mBeep != null) return false;
186        } else if (!mBeep.equals(other.mBeep)) return false;
187        if (mClientSideAssignationDestination == null) {
188            if (other.mClientSideAssignationDestination != null) return false;
189        } else if (!mClientSideAssignationDestination.equals(other.mClientSideAssignationDestination)) return false;
190        if (mDtmfTerm == null) {
191            if (other.mDtmfTerm != null) return false;
192        } else if (!mDtmfTerm.equals(other.mDtmfTerm)) return false;
193        if (mDtmfTermRecognition == null) {
194            if (other.mDtmfTermRecognition != null) return false;
195        } else if (!mDtmfTermRecognition.equals(other.mDtmfTermRecognition)) return false;
196        if (mFinalSilence == null) {
197            if (other.mFinalSilence != null) return false;
198        } else if (!mFinalSilence.equals(other.mFinalSilence)) return false;
199        if (mMaximumTime == null) {
200            if (other.mMaximumTime != null) return false;
201        } else if (!mMaximumTime.equals(other.mMaximumTime)) return false;
202        if (mPostAudioToServer != other.mPostAudioToServer) return false;
203        if (mType == null) {
204            if (other.mType != null) return false;
205        } else if (!mType.equals(other.mType)) return false;
206        return true;
207    }
208
209    @Override
210    public String toString() {
211        return asJson().toString();
212    }
213}
