006import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;
007
008import javax.json.*;
009
010import org.w3c.dom.*;
011
012import com.nuecho.rivr.core.util.*;

017/**
018 * A {@link SupervisedTransfer} is a {@link Transfer} where the outcome is
019 * monitored and where a transfer failure can be handled gracefully.
020 * <p>
021 * Recognition can optionally be activated to cancel a supervised transfer.
022 * 
023 * @author Nu Echo Inc.
024 * @see BridgeTransfer
025 * @see ConsultationTransfer
026 */
027public abstract class SupervisedTransfer extends Transfer {
028    private static final String SPEECH_RECOGNITION_PROPERTY = "speechRecognition";
029    private static final String DTMF_RECOGNITION_PROPERTY = "dtmfRecognition";
030    private static final String TRANSFER_AUDIO_PROPERTY = "transferAudio";
031    private static final String CONNECT_TIMEOUT_PROPERTY = "connectTimeout";
032
033    private DtmfRecognition mDtmfRecognition;
034    private SpeechRecognition mSpeechRecognition;
035
036    private String mTransferAudio;
037    private Duration mConnectTimeout;
038
039    /**
040     * @param name The name of this turn. Not empty.
041     * @param destination The URI of the destination (telephone, IP telephony
042     *            address). Not empty.
043     */
044    public SupervisedTransfer(String name, String destination) {
045        super(name, destination);
046    }
047
048    /**
049     * @param dtmfRecognition The active DTMF recognition configuration during
050     *            the transfer.
051     */
052    public final void setDtmfRecognition(DtmfRecognition dtmfRecognition) {
053        mDtmfRecognition = dtmfRecognition;
054    }
055
056    /**
057     * @param speechRecognition The active speech recognition configuration
058     *            during the transfer.
059     */
060    public final void setSpeechRecognition(SpeechRecognition speechRecognition) {
061        mSpeechRecognition = speechRecognition;
062    }
063
064    /**
065     * @param transferAudio The location (URI or path) of audio source to play
066     *            while the transfer attempt is in progress.
067     */
068    public final void setTransferAudio(String transferAudio) {
069        mTransferAudio = transferAudio;
070    }
071
072    /**
073     * @param connectTimeout The time to wait while trying to connect the call
074     *            before returning with {@link TransferStatus#NO_ANSWER}.
075     *            <code>null</code> to use the VoiceXML platform default.
076     */
077    public final void setConnectTimeout(Duration connectTimeout) {
078        mConnectTimeout = connectTimeout;
079    }
080
081    public final String getTransferAudio() {
082        return mTransferAudio;
083    }
084
085    public final Duration getConnectTimeout() {
086        return mConnectTimeout;
087    }
088
089    public final DtmfRecognition getDtmfRecognition() {
090        return mDtmfRecognition;
091    }
092
093    public final SpeechRecognition getSpeechRecognition() {
094        return mSpeechRecognition;
095    }
096
097    @Override
098    protected void customizeTransferElement(Element transferElement) throws VoiceXmlDocumentRenderingException {
099        VoiceXmlDomUtil.processDtmfRecognition(mDtmfRecognition, transferElement);
100        VoiceXmlDomUtil.processSpeechRecognition(mSpeechRecognition, transferElement);
101        setDurationAttribute(transferElement, CONNECT_TIMEOUT_ATTRIBUTE, mConnectTimeout);
102        setAttribute(transferElement, TRANSFER_AUDIO_ATTRIBUTE, mTransferAudio);
103    }
104
105    @Override
106    protected void addTurnProperties(JsonObjectBuilder builder) {
107        super.addTurnProperties(builder);
108        JsonUtils.addDurationProperty(builder, CONNECT_TIMEOUT_PROPERTY, mConnectTimeout);
109        JsonUtils.add(builder, TRANSFER_AUDIO_PROPERTY, mTransferAudio);
110        JsonUtils.add(builder, DTMF_RECOGNITION_PROPERTY, mDtmfRecognition);
111        JsonUtils.add(builder, SPEECH_RECOGNITION_PROPERTY, mSpeechRecognition);
112    }
113
114    /**
115     * Builder used to ease the creation of instances of
116     * {@link SupervisedTransfer}
117     */
118    public abstract static class Builder extends Transfer.Builder {
119
120        private DtmfRecognition mDtmfRecognition;
121        private SpeechRecognition mSpeechRecognition;
122        private String mTransferAudio;
123        private Duration mConnectTimeout;
124
125        protected Builder(String name) {
126            super(name);
127        }
128
129        public Builder setDtmfRecognition(DtmfRecognition dtmfRecognition) {
130            mDtmfRecognition = dtmfRecognition;
131            return this;
132        }
133
134        public Builder setSpeechRecognition(SpeechRecognition speechRecognition) {
135            mSpeechRecognition = speechRecognition;
136            return this;
137        }
138
139        public Builder setTransferAudio(String transferAudio) {
140            mTransferAudio = transferAudio;
141            return this;
142        }
143
144        public Builder setConnectTimeout(Duration connectTimeout) {
145            mConnectTimeout = connectTimeout;
146            return this;
147        }
148
149        public void build(SupervisedTransfer supervisedTransfer) {
150            supervisedTransfer.setConnectTimeout(mConnectTimeout);
151            supervisedTransfer.setDtmfRecognition(mDtmfRecognition);
152            supervisedTransfer.setSpeechRecognition(mSpeechRecognition);
153            supervisedTransfer.setTransferAudio(mTransferAudio);
154            super.build(supervisedTransfer);
155        }
156    }
157
158    @Override
159    public int hashCode() {
160        final int prime = 31;
161        int result = super.hashCode();
162        result = prime * result + ((mConnectTimeout == null) ? 0 : mConnectTimeout.hashCode());
163        result = prime * result + ((mDtmfRecognition == null) ? 0 : mDtmfRecognition.hashCode());
164        result = prime * result + ((mSpeechRecognition == null) ? 0 : mSpeechRecognition.hashCode());
165        result = prime * result + ((mTransferAudio == null) ? 0 : mTransferAudio.hashCode());
166        return result;
167    }
168
169    @Override
170    public boolean equals(Object obj) {
171        if (this == obj) return true;
172        if (!super.equals(obj)) return false;
173        if (getClass() != obj.getClass()) return false;
174        SupervisedTransfer other = (SupervisedTransfer) obj;
175        if (mConnectTimeout == null) {
176            if (other.mConnectTimeout != null) return false;
177        } else if (!mConnectTimeout.equals(other.mConnectTimeout)) return false;
178        if (mDtmfRecognition == null) {
179            if (other.mDtmfRecognition != null) return false;
180        } else if (!mDtmfRecognition.equals(other.mDtmfRecognition)) return false;
181        if (mSpeechRecognition == null) {
182            if (other.mSpeechRecognition != null) return false;
183        } else if (!mSpeechRecognition.equals(other.mSpeechRecognition)) return false;
184        if (mTransferAudio == null) {
185            if (other.mTransferAudio != null) return false;
186        } else if (!mTransferAudio.equals(other.mTransferAudio)) return false;
187        return true;
188    }
