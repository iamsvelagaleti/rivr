005package com.nuecho.rivr.voicexml.servlet;
006
007import java.util.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.voicexml.util.json.*;
012
013/**
014 * HTTP-uploaded file via <code>multipart/form-data</code>, typically the audio
015 * file of a recording. This class contains the content type, the headers and
016 * the data (ie. a byte array).
017 * 
018 * @author Nu Echo Inc.
019 */
020public final class FileUpload implements JsonSerializable {
021
022    private static final String SIZE_PROPERTY = "size";
023    private static final String TYPE_PROPERTY = "type";
024    private static final String FILENAME_PROPERTY = "filename";
025
026    private final String mName;
027    private final String mContentType;
028    private final byte[] mContent;
029    private final Map<String, String> mHeaders;
030
031    public FileUpload(String name, String contentType, byte[] content, Map<String, String> headers) {
032        mName = name;
033        mContentType = contentType;
034        mContent = content;
035        mHeaders = headers;
036    }
037
038    public String getName() {
039        return mName;
040    }
041
042    public String getContentType() {
043        return mContentType;
044    }
045
046    public byte[] getContent() {
047        return mContent;
048    }
049
050    public Set<String> getHeaderNames() {
051        return Collections.unmodifiableSet(mHeaders.keySet());
052    }
053
054    public String getHeader(String name) {
055        return mHeaders.get(name);
056    }
057
058    @Override
059    public String toString() {
060        return asJson().toString();
061    }
062
063    @Override
064    public JsonValue asJson() {
065        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
066        JsonUtils.add(builder, FILENAME_PROPERTY, mName);
067        JsonUtils.add(builder, TYPE_PROPERTY, mContentType);
068        if (mContent != null) {
069            builder.add(SIZE_PROPERTY, (long) mContent.length);
070        }
071        //content is not serialized
072        return builder.build();
073    }
074
075    @Override
076    public int hashCode() {
077        final int prime = 31;
078        int result = 1;
079        result = prime * result + ((mContentType == null) ? 0 : mContentType.hashCode());
080        result = prime * result + Arrays.hashCode(mContent);
081        result = prime * result + ((mHeaders == null) ? 0 : mHeaders.hashCode());
082        result = prime * result + ((mName == null) ? 0 : mName.hashCode());
083        return result;
084    }
085
086    @Override
087    public boolean equals(Object obj) {
088        if (this == obj) return true;
089        if (obj == null) return false;
090        if (getClass() != obj.getClass()) return false;
091        FileUpload other = (FileUpload) obj;
092        if (mContentType == null) {
093            if (other.mContentType != null) return false;
094        } else if (!mContentType.equals(other.mContentType)) return false;
095        if (!Arrays.equals(mContent, other.mContent)) return false;
096        if (mHeaders == null) {
097            if (other.mHeaders != null) return false;
098        } else if (!mHeaders.equals(other.mHeaders)) return false;
099        if (mName == null) {
100            if (other.mName != null) return false;
101        } else if (!mName.equals(other.mName)) return false;
102        return true;
103    }
104}
